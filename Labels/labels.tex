\begin{luacode*}

  local identifiers = {"A", "B"}
  
  for _, year in ipairs(years) do
    for _, identifier in ipairs(identifiers) do
      for number = 1, 64 do
        local id = string.format("%s-%s-%03d", identifier, year, number)
        tex.sprint("\\addresslabel{\\normalsize{" .. id .. "} \\qrcode{doc-id:" .. id .. "}}")
      end
    end
  end
